import { useState } from "react";

const RING_RADIUS = 40;
const CIRCUMFERENCE = 2 * Math.PI * RING_RADIUS;

function ScoreRing({ value, color, label, icon }) {
  const pct = Math.min(Math.max(value, 0), 1);
  const dash = pct * CIRCUMFERENCE;
  const gap = CIRCUMFERENCE - dash;

  return (
    <div style={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 8 }}>
      <svg width="96" height="96" viewBox="0 0 96 96">
        <circle cx="48" cy="48" r={RING_RADIUS} fill="none" stroke="var(--color-border-tertiary)" strokeWidth="7" />
        <circle
          cx="48" cy="48" r={RING_RADIUS}
          fill="none"
          stroke={color}
          strokeWidth="7"
          strokeDasharray={`${dash} ${gap}`}
          strokeLinecap="round"
          transform="rotate(-90 48 48)"
          style={{ transition: "stroke-dasharray 0.6s cubic-bezier(.4,0,.2,1)" }}
        />
        <text x="48" y="44" textAnchor="middle" fontSize="11" fill="var(--color-text-secondary)" fontFamily="var(--font-sans)">{icon}</text>
        <text x="48" y="60" textAnchor="middle" fontSize="16" fontWeight="500" fill="var(--color-text-primary)" fontFamily="var(--font-sans)">
          {Math.round(pct * 100)}%
        </text>
      </svg>
      <span style={{ fontSize: 12, color: "var(--color-text-secondary)", letterSpacing: "0.03em" }}>{label}</span>
    </div>
  );
}

function Badge({ verdict }) {
  const map = {
    likely_fake:   { bg: "#FCEBEB", text: "#A32D2D", label: "Likely fake"    },
    suspicious:    { bg: "#FAEEDA", text: "#854F0B", label: "Suspicious"      },
    likely_genuine:{ bg: "#EAF3DE", text: "#3B6D11", label: "Likely genuine" },
  };
  const s = map[verdict] || map.suspicious;
  return (
    <span style={{
      background: s.bg,
      color: s.text,
      fontSize: 12,
      fontWeight: 500,
      padding: "3px 10px",
      borderRadius: 99,
      letterSpacing: "0.04em",
    }}>{s.label}</span>
  );
}

function verdictFromScores(fake, trust) {
  if (fake > 0.65) return "likely_fake";
  if (fake > 0.35 || trust < 0.55) return "suspicious";
  return "likely_genuine";
}

export default function FakeReviewDetector() {
  const [text, setText] = useState("");
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState(null);
  const [error, setError] = useState("");

  async function analyzeReview() {
    if (!text.trim()) return;
    setLoading(true);
    setResult(null);
    setError("");

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          messages: [{
            role: "user",
            content: `You are a fake review detection expert. Analyze the following review and respond with ONLY a JSON object (no markdown, no code fences):
{
  "fake_probability": <number 0-1>,
  "trust_score": <number 0-1>,
  "explanation": [<3-5 concise bullet strings>]
}

Review:
"""
${text}
"""`,
          }],
        }),
      });

      const data = await res.json();
      const raw = data.content?.map(b => b.text || "").join("").trim();
      const clean = raw.replace(/```json|```/g, "").trim();
      const parsed = JSON.parse(clean);
      setResult(parsed);
    } catch (e) {
      setError("Analysis failed. Please try again.");
    } finally {
      setLoading(false);
    }
  }

  const verdict = result ? verdictFromScores(result.fake_probability, result.trust_score) : null;

  return (
    <div style={{
      minHeight: "100vh",
      background: "var(--color-background-tertiary)",
      display: "flex",
      alignItems: "flex-start",
      justifyContent: "center",
      padding: "2.5rem 1rem",
    }}>
      <div style={{
        width: "100%",
        maxWidth: 560,
        display: "flex",
        flexDirection: "column",
        gap: "1.25rem",
      }}>

        {/* Header */}
        <div style={{ marginBottom: "0.25rem" }}>
          <h1 style={{ fontSize: 22, fontWeight: 500, margin: 0, color: "var(--color-text-primary)" }}>
            Fake review detector
          </h1>
          <p style={{ fontSize: 14, color: "var(--color-text-secondary)", margin: "4px 0 0" }}>
            Paste any product or service review to analyse its authenticity.
          </p>
        </div>

        {/* Input card */}
        <div style={{
          background: "var(--color-background-primary)",
          border: "0.5px solid var(--color-border-tertiary)",
          borderRadius: "var(--border-radius-lg)",
          padding: "1rem 1.25rem",
          display: "flex",
          flexDirection: "column",
          gap: "0.75rem",
        }}>
          <textarea
            value={text}
            onChange={e => setText(e.target.value)}
            placeholder="Paste a review here…"
            style={{
              width: "100%",
              minHeight: 120,
              resize: "vertical",
              border: "0.5px solid var(--color-border-secondary)",
              borderRadius: "var(--border-radius-md)",
              padding: "10px 12px",
              fontSize: 14,
              fontFamily: "var(--font-sans)",
              color: "var(--color-text-primary)",
              background: "var(--color-background-secondary)",
              boxSizing: "border-box",
              outline: "none",
              lineHeight: 1.6,
            }}
          />
          <button
            onClick={analyzeReview}
            disabled={loading || !text.trim()}
            style={{
              alignSelf: "flex-end",
              padding: "8px 20px",
              fontSize: 14,
              fontWeight: 500,
              fontFamily: "var(--font-sans)",
              background: loading || !text.trim() ? "var(--color-background-secondary)" : "var(--color-text-primary)",
              color: loading || !text.trim() ? "var(--color-text-tertiary)" : "var(--color-background-primary)",
              border: "0.5px solid var(--color-border-secondary)",
              borderRadius: "var(--border-radius-md)",
              cursor: loading || !text.trim() ? "not-allowed" : "pointer",
              transition: "opacity 0.15s",
            }}
          >
            {loading ? "Analysing…" : "Analyse review"}
          </button>
        </div>

        {error && (
          <p style={{ fontSize: 13, color: "var(--color-text-danger)", margin: 0 }}>{error}</p>
        )}

        {/* Results */}
        {result && (
          <div style={{
            background: "var(--color-background-primary)",
            border: "0.5px solid var(--color-border-tertiary)",
            borderRadius: "var(--border-radius-lg)",
            padding: "1.25rem",
            display: "flex",
            flexDirection: "column",
            gap: "1.25rem",
          }}>

            {/* Verdict row */}
            <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between" }}>
              <p style={{ margin: 0, fontSize: 13, color: "var(--color-text-secondary)" }}>Verdict</p>
              <Badge verdict={verdict} />
            </div>

            {/* Score rings */}
            <div style={{
              display: "flex",
              justifyContent: "space-around",
              padding: "0.5rem 0",
              borderTop: "0.5px solid var(--color-border-tertiary)",
              borderBottom: "0.5px solid var(--color-border-tertiary)",
            }}>
              <ScoreRing
                value={result.fake_probability}
                color="#E24B4A"
                label="Fake probability"
                icon="!"
              />
              <ScoreRing
                value={result.trust_score}
                color="#639922"
                label="Trust score"
                icon="✓"
              />
            </div>

            {/* Explanation */}
            <div>
              <p style={{ margin: "0 0 10px", fontSize: 13, color: "var(--color-text-secondary)", fontWeight: 500 }}>
                Signals detected
              </p>
              <ul style={{ margin: 0, padding: 0, listStyle: "none", display: "flex", flexDirection: "column", gap: 6 }}>
                {result.explanation.map((item, i) => (
                  <li key={i} style={{
                    fontSize: 13,
                    color: "var(--color-text-primary)",
                    background: "var(--color-background-secondary)",
                    padding: "8px 12px",
                    borderRadius: "var(--border-radius-md)",
                    lineHeight: 1.5,
                    borderLeft: "2px solid var(--color-border-secondary)",
                  }}>
                    {item}
                  </li>
                ))}
              </ul>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
